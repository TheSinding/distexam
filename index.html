              import { mapActions } from 'vuex'
              export default {
                name: 'PageIndex',
                data: () => ({ name: '' }),
                computed: {
                  disabled() {
                    return this.name === ''
                  }
                },
                methods: {
                  ...mapActions('game', ['join']),
                  joinGame() {
                    this.join({ name: this.name })
                    this.$router.push({ name: 'lobby' })
                  }
                }
              }
